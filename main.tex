\section{\textbf{Question}}
\hfill(GATE PH 2023)
\begin{enumerate}
    \item  Which one of the following options is CORRECT for the given logic circuit?
\end{enumerate}
\vspace{1.0mm}
\begin{enumerate}[label=(\alph*)]
    \item P = 1, Q = 1; X = 0
    \item P = 1, Q = 0; X = 0
    \item P = 0, Q = 1; X = 0
    \item P = 0, Q = 0; X = 1
\end{enumerate}
\hfill{GATE IN 2023}
    \section{\textbf{Answer}}
    $\rightarrow P'+ Q$\\
    Therefore, the Boolean function $F(X)=(P' + Q)$
